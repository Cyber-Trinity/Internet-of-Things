# ARDUINO FIRMWARE DEPENDENCIES

SoftwareSerial.h # Enables serial communication between the Arduino (I/O) and the ESP32.
Keypad.h # Read and process input from the physical 4x3 Keypad for OTP entry.
Servo.h # Control the Servo Motor, which manages the physical door locking mechanism.
LiquidCrystal_I2C.h # Control and display messages on the LCD screen for user prompts.
ArduinoJson.h # Parse and generate JSON data sent over the serial connection.


# ESP32 FIRMWARE DEPENDENCIES

WiFi.h # Essential for connecting the ESP32 to the Wi-Fi network.
HTTPClient.h # Used to make secure HTTPS (REST API) requests to the Supabase cloud backend.
ArduinoJson.h # Used for creating JSON payloads for API calls and parsing JSON responses from Supabase.
HardwareSerial.h # Used for high-speed serial communication with the Arduino UNO.


# WEB APPLICATION DEPENDENCIES (NEXT.JS)

next # The core React framework for server-side rendering the web dashboard.
react # The library for building the user interface components.
react-dom # Provides the method for DOM-specific rendering.
tailwindcss # A utility-first CSS framework for responsive styling of the dashboard.
autoprefixer # A PostCSS plugin to add vendor prefixes to CSS rules.
postcss # A tool for transforming CSS with JavaScript plugins.
clsx # Utility for conditionally combining CSS class names.
@radix-ui/react-slot # Utility for flexible component composition.
@supabase/supabase-js # Client library for interacting with the Supabase database and authentication.
date-fns # A simple library for working with dates and times.
lucide-react # Provides high-quality vector icons for the user interface.
